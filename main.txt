#include <stdio.h>
#include <math.h>
int main()
{ 
    int dim,count,parne_2,parne_2_pos;
    scanf("%d",&dim);
    int arr[dim];
    float arr_mod[dim];
    for(int i=0;i<dim;i++){
        scanf("%d",&arr[i]);
    }
    for(int i=0;i<dim;i++)
    { 
        if(arr[i]%2 == 0)
        {   count++;
            if(count==2)
                parne_2=arr[i];
                parne_2_pos=i;
        }
     if(count==2) break;
    }
    for(int i=0;i<dim;i++)
    {   arr_mod[i]=arr[i];
        if(i>parne_2_pos) arr_mod[i]=arr_mod[i]/parne_2;
    }
    if(count==0)
        printf("В масиві немає парних елементів\n");
    if (count==1)
        printf("В масиві тільки один парий елемент\n");
    return 0;
}
